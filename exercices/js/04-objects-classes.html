Solution:
function Car(brand, model, year) { this.brand = brand; this.model = model; this.year = year; this.getInfo = function() { return `${this.year} ${this.brand} ${this.model}`; }; this.isOld = function() { let currentYear = new Date().getFullYear(); return (currentYear - this.year) > 10; }; } let car1 = new Car("Toyota", "Camry", 2020); let car2 = new Car("Ford", "Mustang", 2010); console.log(car1.getInfo()); console.log("Is old:", car1.isOld()); console.log(car2.getInfo()); console.log("Is old:", car2.isOld());

Exercise 4: Classes

Create classes using modern ES6 syntax.

// TODO: Create a class called 'Student' with properties: name, age, grade // TODO: Add a constructor method // TODO: Add a method called 'getInfo' that returns student information // TODO: Add a method called 'isPassing' that returns true if grade >= 60 // TODO: Add a method called 'study' that increases grade by 5 points // TODO: Create student instances and test all methods
Solution:
class Student { constructor(name, age, grade) { this.name = name; this.age = age; this.grade = grade; } getInfo() { return `${this.name} (${this.age} years old) - Grade: ${this.grade}`; } isPassing() { return this.grade >= 60; } study() { this.grade += 5; if (this.grade > 100) this.grade = 100; return `${this.name} studied and now has grade ${this.grade}`; } } let student1 = new Student("Alice", 20, 75); let student2 = new Student("Bob", 19, 55); console.log(student1.getInfo()); console.log("Is passing:", student1.isPassing()); console.log(student1.study()); console.log(student2.getInfo()); console.log("Is passing:", student2.isPassing()); console.log(student2.study());

Exercise 6: Inheritance

Practice class inheritance and the super keyword.

// TODO: Create a base class called 'Animal' with properties: name, age // TODO: Add a method called 'speak' that returns "[name] makes a sound" // TODO: Create a subclass called 'Dog' that extends Animal // TODO: Add a property 'breed' to Dog // TODO: Override the speak method to return "[name] barks" // TODO: Create a subclass called 'Cat' that extends Animal // TODO: Override the speak method to return "[name] meows" // TODO: Create instances and test all methods
Solution:
class Animal { constructor(name, age) { this.name = name; this.age = age; } speak() { return `${this.name} makes a sound`; } } class Dog extends Animal { constructor(name, age, breed) { super(name, age); this.breed = breed; } speak() { return `${this.name} barks`; } } class Cat extends Animal { constructor(name, age) { super(name, age); } speak() { return `${this.name} meows`; } } let dog = new Dog("Buddy", 5, "Golden Retriever"); let cat = new Cat("Whiskers", 3); console.log(dog.speak()); console.log("Breed:", dog.breed); console.log(cat.speak());

Solution:
class BankAccount { constructor(initialBalance = 0) { this._balance = initialBalance; } get balance() { return this._balance; } set balance(value) { if (value >= 0) { this._balance = value; } else { console.log("Balance cannot be negative"); } } deposit(amount) { if (amount > 0) { this._balance += amount; return `Deposited $${amount}. New balance: $${this._balance}`; } else { return "Invalid deposit amount"; } } withdraw(amount) { if (amount > 0 && amount <= this._balance) { this._balance -= amount; return `Withdrew $${amount}. New balance: $${this._balance}`; } else { return "Insufficient funds or invalid amount"; } } } let account = new BankAccount(100); console.log("Initial balance:", account.balance); console.log(account.deposit(50)); console.log(account.withdraw(30)); console.log(account.withdraw(200)); // Should fail
