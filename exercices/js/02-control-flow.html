Exercise 2: Switch Statement

Create a program that converts numbers to day names.

// TODO: Create a variable 'dayNumber' with a number (1-7) // TODO: Use switch statement to convert: // 1 = "Monday", 2 = "Tuesday", 3 = "Wednesday" // 4 = "Thursday", 5 = "Friday", 6 = "Saturday", 7 = "Sunday" // default = "Invalid day" // TODO: Print the day name using console.log()
Solution:
let dayNumber = 3; let dayName; switch(dayNumber) { case 1: dayName = "Monday"; break; case 2: dayName = "Tuesday"; break; case 3: dayName = "Wednesday"; break; case 4: dayName = "Thursday"; break; case 5: dayName = "Friday"; break; case 6: dayName = "Saturday"; break; case 7: dayName = "Sunday"; break; default: dayName = "Invalid day"; } console.log(dayName);

Exercise 3: While Loop

Create a countdown program using a while loop.

// TODO: Create a variable 'countdown' starting at 10 // TODO: Use while loop to count down from 10 to 1 // TODO: Print each number using console.log() // TODO: After the loop, print "Blast off!"
Solution:
let countdown = 10; while (countdown > 0) { console.log(countdown); countdown--; } console.log("Blast off!");

Exercise 4: For Loop

Calculate the sum of numbers from 1 to 100 using a for loop.

// TODO: Create a variable 'sum' starting at 0 // TODO: Use for loop to iterate from 1 to 100 // TODO: Add each number to the sum // TODO: Print the final sum using console.log()
Solution:
let sum = 0; for (let i = 1; i <= 100; i++) { sum += i; } console.log("Sum of numbers from 1 to 100:", sum);

Exercise 5: Interactive Number Guessing Game

Create a simple number guessing game.

// TODO: Generate a random number between 1 and 10 // TODO: Get user input (you can use prompt() or DOM) // TODO: Compare the guess with the random number // TODO: Give feedback: "Too high", "Too low", or "Correct!" // TODO: Keep track of attempts
Solution:
let targetNumber = Math.floor(Math.random() * 10) + 1; let attempts = 0; function checkGuess() { let guess = parseInt(document.getElementById("guessInput").value); attempts++; if (guess < targetNumber) { document.getElementById("gameResult").textContent = "Too low!"; } else if (guess > targetNumber) { document.getElementById("gameResult").textContent = "Too high!"; } else { document.getElementById("gameResult").textContent = "Correct! It took you " + attempts + " attempts."; } } function newGame() { targetNumber = Math.floor(Math.random() * 10) + 1; attempts = 0; document.getElementById("guessInput").value = ""; document.getElementById("gameResult").textContent = ""; }

Exercise 7: Temperature Converter

Create a temperature conversion program with user input.

// TODO: Get temperature value from input // TODO: Get conversion type from select // TODO: Use if/else to handle both conversions: // Celsius to Fahrenheit: (C * 9/5) + 32 // Fahrenheit to Celsius: (F - 32) * 5/9 // TODO: Display the result
Solution:
function convertTemperature() { let temp = parseFloat(document.getElementById("tempInput").value); let conversionType = document.getElementById("conversionType").value; let result; if (conversionType === "c2f") { result = (temp * 9/5) + 32; document.getElementById("tempResult").textContent = temp + "°C = " + result.toFixed(1) + "°F"; } else { result = (temp - 32) * 5/9; document.getElementById("tempResult").textContent = temp + "°F = " + result.toFixed(1) + "°C"; } }
