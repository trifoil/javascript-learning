    Exercise 5: Todo List Application

    Create a complete todo list with add, complete, and delete functionality.

    // TODO: Create an array to store todo items // TODO: Function to add new todo items // TODO: Function to mark items as complete // TODO: Function to delete items // TODO: Function to display all todos // TODO: Add event listeners for complete and delete buttons
    Solution:
    let todos = []; function addTodo() { let input = document.getElementById("todoInput"); let text = input.value.trim(); if (text !== "") { let todo = { id: Date.now(), text: text, completed: false }; todos.push(todo); input.value = ""; displayTodos(); } } function toggleTodo(id) { let todo = todos.find(t => t.id === id); if (todo) { todo.completed = !todo.completed; displayTodos(); } } function deleteTodo(id) { todos = todos.filter(t => t.id !== id); displayTodos(); } function displayTodos() { let todoList = document.getElementById("todoList"); todoList.innerHTML = ""; todos.forEach(todo => { let todoItem = document.createElement("div"); todoItem.className = "todo-item"; if (todo.completed) { todoItem.classList.add("completed"); } todoItem.innerHTML = ` ${todo.text}
    `; todoList.appendChild(todoItem); }); }

    Exercise 6: Form Validation

    Create a form with real-time validation.

    // TODO: Add input event listeners for real-time validation // TODO: Validate name (not empty, at least 2 characters) // TODO: Validate email (contains @ and .) // TODO: Validate age (between 18 and 120) // TODO: Show/hide error messages // TODO: Prevent form submission if validation fails
    Solution:
    let form = document.getElementById("validationForm"); let nameInput = document.getElementById("nameInput"); let emailInput = document.getElementById("emailInput"); let ageInput = document.getElementById("ageInput"); // Real-time validation nameInput.addEventListener("input", validateName); emailInput.addEventListener("input", validateEmail); ageInput.addEventListener("input", validateAge); function validateName() { let name = nameInput.value.trim(); let error = document.getElementById("nameError"); if (name.length < 2) { error.textContent = "Name must be at least 2 characters long"; error.style.display = "block"; return false; } else { error.style.display = "none"; return true; } } function validateEmail() { let email = emailInput.value.trim(); let error = document.getElementById("emailError"); let emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/; if (!emailRegex.test(email)) { error.textContent = "Please enter a valid email address"; error.style.display = "block"; return false; } else { error.style.display = "none"; return true; } } function validateAge() { let age = parseInt(ageInput.value); let error = document.getElementById("ageError"); if (age < 18 || age > 120 || isNaN(age)) { error.textContent = "Age must be between 18 and 120"; error.style.display = "block"; return false; } else { error.style.display = "none"; return true; } } form.addEventListener("submit", function(event) { event.preventDefault(); if (validateName() && validateEmail() && validateAge()) { document.getElementById("formResult").textContent = "Form submitted successfully!"; form.reset(); } else { document.getElementById("formResult").textContent = "Please fix the errors above."; } });
